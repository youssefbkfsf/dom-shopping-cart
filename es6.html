// ES6 class definition class Person { // Constructor to initialize object properties constructor(name, age) { this.name = name; this.age = age; } // Method to display information about the person displayInfo() { console.log(`Name: ${this.name}, Age: ${this.age}`); } } // Creating instances of the Person class const person1 = new Person("John Doe", 25); const person2 = new Person("Jane Smith", 30); // Calling the displayInfo method on each instance person1.displayInfo(); person2.displayInfo();
